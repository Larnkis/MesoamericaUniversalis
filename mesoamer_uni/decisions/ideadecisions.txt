#These decisions require ideas as prerequesites
country_decisions = {
	#Hold Grand Festival
	hold_grand_festival = {
		potential = {
			has_idea = grand_spectacle 
		}
		
		allow = {
			NOT = { has_country_modifier = grand_festival_CD }
			years_of_income = 0.75
		}
		
		effect = {
			add_country_modifier = {
				name = grand_festival_CD
				duration = 3650
			}
			add_years_of_income = -0.75
			country_event = { id = sideaevents.01 }
		}
	}
	
	#Resettle to Colonies
	resettle_colonies = {
		potential = {
			has_idea = resettlement
		}
		
		allow = {
			NOT = { has_country_modifier = has_resettled }
			any_province = {
				AND = {
					is_colony = yes
					owned_by = ROOT
				}
			}
		}
		
		effect = {
			add_country_modifier = {
				name = has_resettled 
				duration = 3650
			}
			every_province = {
				limit = {
					is_colony = yes
					owned_by = ROOT
				}
				add_colonysize = 250
				add_province_modifier = {
					name = resettled_colonists
					duration = 1825
				}
			}
		}
	}
	
	
}
