country_decisions = {
	#Upgrade - Duchy to Kingdom
	upgrade_duchy = {
		major = yes
		potential = {
			government_rank = 1
			NOT = { government_rank = 2 }
		}
		allow = {
			prestige = 25
			num_of_cities = 20
		}
		effect = {
			set_government_rank = 2
			add_prestige = 10
			add_country_modifier = { 
				name = upgraded_government
				duration = 1825 
			}
			add_country_modifier = {
				name = baby_boom_global
				duration = 1825
			}
		}
	}
	#Upgrade - Kingdom to Empire
	upgrade_kingdom = {
		major = yes
		potential = {
			government_rank = 2
			NOT = { government_rank = 3 }
			NOT = { 
				OR = { 
					tag = TEN
					tag = TET
					tag = TCP
					tag = IUA
					tag = PAT
					tag = IHU
				}
			}
		}
		allow = {
			prestige = 75
			num_of_cities = 50
		}
		effect = {
			set_government_rank = 3
			add_prestige = 25
			add_country_modifier = { 
				name = upgraded_government
				duration = 1825 
			}add_country_modifier = {
				name = baby_boom_global
				duration = 1825
			}
		}
	}
	#Switch - Territorial
	switch_territorial = {
		major = yes
		potential = {
			NOT = { government = centralized_altepetl }
		}
		allow = {
			has_idea = territorial_ideology 
			adm_power = 200
		}
		effect = {
			add_adm_power = -200
			change_government = centralized_altepetl 
			if = {
				limit = {
					NOT = { 
						check_variable = {
							which = centralization 
							value = 10
						}
					}
				}
				set_variable = { which = centralization value = 10 }
			}
			remove_country_modifier = tribreqs_normal
		}
	}
	#Switch - Hegemonic
	switch_hegemonic = {
		major = yes
		potential = {
			NOT = { government = altepetl }
		}
		allow = {
			has_idea = hegemonic_ideology 
			adm_power = 200
		}
		effect = {
			add_adm_power = -200
			change_government = altepetl 
			if = {
				limit = {
					check_variable = {
						which = centralization 
						value = -40
					}
				}
				set_variable = { which = centralization value = -40 }
			}
			add_country_modifier = {
				name = "tribreqs_normal"
				duration = -1
			}
		}
	}
}
