country_decisions = {

	buydowninfluence = {
		potential = {
			always = no
			NOT = { has_country_flag = bdiopen } 
		}
		
		allow = {
			
		}

			set_country_flag = bdiopen
			country_event = { id = bdi.01 }
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	#Exchange Luxury Goods for Ducats
	exchangeluxurygoods = {
		potential = {
			
		}
	
		allow = {
			check_variable = { which = luxurygoods value = 25} 
		}

		effect = {
			add_treasury = 100
			change_variable = { which = luxurygoods value = -25 }
		}

		ai_will_do = {
			factor = 0

				factor = 1000
				check_variable = { which = luxurygoods value = 25 }
			}
		}
	}

		}
