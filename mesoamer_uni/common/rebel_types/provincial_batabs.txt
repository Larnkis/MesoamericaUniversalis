	independence = none				# Does not strive for independence

	# The rebel type with the highest modifier for this province gets picked

		factor = 10

					primary_culture = yucatecmayan

				}
			}
		}

			factor = 0.5
			nationalism = 10
		}
		modifier = {
			factor = 1.25
			NOT = { nationalism = 10 }
		}
		modifier = {
			factor = 1.50
			NOT = { nationalism = 5 }
		}
		modifier = {
			factor = 1.75
			NOT = { nationalism = 1 }
		}
		
		modifier = {
			factor = 1.25
			owner = {
				check_variable = { which = centralization value = 25 }
			}
		}
		
		modifier = {
			factor = 1.50
			owner = {

				check_variable = { which = centralization value = 75 }

		}

		factor = 1

 	siege_won_trigger = {

	can_negotiate_trigger = {
