#Code a scripted effect that determines if a country is territorial or hegemonic

territorial_or_hegemonic = {
	clr_country_flag = territorial_state
	clr_country_flag = tributary_state
	set_country_flag = territorial_state
	if = {
		limit = { 
			OR = {  
				government = altepetl
				government = tlaxcallan_confederate
			}
		}
		clr_country_flag = territorial_state
		set_country_flag = tributary_state
	}
}

change_master = {
	set_variable = { which = admin_influence_overlord value = 0 }
	set_variable = { which = subjectrevoltrisk value = 0 }
}

merchants_leave = {
	clr_country_flag = ms_repeal_sumptuary_laws
	clr_country_flag = ms_permit_land_ownership
    clr_country_flag = ms_oppress_foreign_merchants
	clr_country_flag = ms_unlimited_trade
	clr_country_flag = ms_guarantee_legal_protection
	subtract_variable = { which = ms_favors which = ms_favors }
}
