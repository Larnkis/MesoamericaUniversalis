#Triggered Modifiers from Power Perception

pwrper_0_10 = {
	potential = {
		always = yes
	}
	
	trigger = {
		AND = { 
			check_variable = { which = power_perception value = 0 }
			NOT = { check_variable = { which = power_perception value = 10 } }
		}
	}
	
	global_unrest = 20
	reduced_liberty_desire = -100
}
