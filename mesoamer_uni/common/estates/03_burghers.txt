	icon = 3

	# Is estate kept in control of province on conquest?
	keep_on_conquest = no	
	
	min_provinces_to_want_territory = 20
	wanted_territory_modifier = {
		desc = EST_VAL_REPUBLIC
		trigger = {
			government = republic
			NOT = { has_disaster = estate_nobility_disaster }
		}
		value = 20
	}	

	# If true, country will get estate
	trigger = {	
		NOT = { government = merchant_republic }
		NOT = { government = native_council }
		NOT = { government = siberian_native_council }
		is_nomad = no
		is_colonial_nation = no
		NOT = { government = celestial_empire }
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		OR = {
			owner = { government = republic }
			development = 10
			province_trade_power = 5
			owner = { government = republic }
		}
	}

	min_autonomy = 50

	country_modifier_happy = {
		trade_efficiency = 0.33
		trade_steering = 0.33
	}
	country_modifier_neutral = {
		trade_efficiency = 0.33
	}	
	country_modifier_angry = {
		trade_efficiency = -0.33
		trade_steering = -0.33
	}

	province_modifier_happy = {
		trade_goods_size_modifier = 0.1

	}

		province_trade_power_modifier = 0.5

		local_unrest = 4

	influence_modifier = {	
		desc = EST_HIGH_DEVELOPMENT_PROVINCE
		trigger = {
			any_owned_province = {
				development = 4000
			}
		}
		influence = 5
	}	
	
	influence_modifier = {
		desc = EST_VAL_LOW_TRADE_INCOME
		trigger = {
			NOT = { trade_income_percentage = 0.25 }
		}	
		influence = -5
	}
	
	influence_modifier = {
		desc = EST_VAL_HIGH_TRADE_INCOME
		trigger = {
			trade_income_percentage = 0.5
			NOT = { trade_income_percentage = 0.75 }
		}	
		influence = 5
	}

		desc = EST_VAL_VERY_HIGH_TRADE_INCOME
		trigger = {
			trade_income_percentage = 0.75
		}	
		influence = 10
	}

		trigger = {

		}

	}

			NOT = { religion_group = christian } #Diversify a bit more.

	}

	ai_grant_province = {
		factor = 1
		modifier = {
			factor = 0.5
			NOT = { province_trade_power = 5 }
		}			
		modifier = {
			factor = 2
			province_trade_power = 10
		}
		modifier = {
			factor = 0.1

	}

	ai_territory_modifier = {

		icon = 6

		potential = {

			always = yes				
		}
		
		effect = {

		}

			factor = 0

		icon = 6
		
		years_between_use = 0
		
		potential = {
			always = yes

			always = yes

				subtract_variable = { which = blcmerchantshigh which = blcmerchantshigh }
				change_variable = { which = blcmerchantshigh which = blcmerchants }

				multiply_variable = { which = blcmerchants value = 3 }

			country_event = { id = blg.03 }
		}
		
		ai_will_do = {
			factor = 0		
		}

		name = "FM_ESP"
		
		icon = 6
		
		years_between_use = 15
		
		potential = {
			always = yes
		}	
		
		trigger = {

				estate = estate_burghers
				loyalty = 60
			}
			estate_influence = {
				estate = estate_burghers 
				influence = 25
			}
		}
		
		effect = {
			add_estate_loyalty = {
				estate = estate_burghers
				loyalty = -5

				name = crazyspies
				duration = 1825
			}
		}

			always = yes

				name = weakenedmerchants

		name = "FM_SET"
		
		icon = 6
		
		years_between_use = 10
		
		potential = {
			always = yes
		}	
		
		trigger = {
			AND = {
			estate_loyalty = {
				estate = estate_burghers
				loyalty = 60

				estate = estate_burghers 
				influence = 45
			}
			OR = {
				any_country = {

			add_estate_loyalty = {
				estate = estate_burghers
				loyalty = -10
			}

					name = sabotagedtrade
