namespace = estatetribute

#From Nobles
country_event = {
	id = estatetribute.01

	title = "estatetribute1.t"
	desc = "estatetribute1.d"
	picture = TRADEGOODS_eventPicture

	is_triggered_only = yes
	
	#Take the tribute
	option = {
		name = "estatetribute1.a"
		trigger = {
			always = yes
		}
		
		change_variable = { which = luxurygoods which = luxurygoods_n }
		change_variable = { which = staplefooditems which = staplefooditems_n }
		change_variable = { which = buildingsupplies which = buildingsupplies_n }
		change_variable = { which = militarysupplies which = militarysupplies_n }
		
		hidden_effect = {
			subtract_variable = { which = luxurygoods_n which = luxurygoods_n }
			subtract_variable = { which = staplefooditems_n which = staplefooditems_n }
			subtract_variable = { which = buildingsupplies_n which = buildingsupplies_n }
			subtract_variable = { which = militarysupplies_n which = militarysupplies_n }
		}
	}
	
	#Exempt them as promised
	option = {
		name = "estatetribute1.b"
		trigger = {
			always = no
		}
		hidden_effect = {
			subtract_variable = { which = luxurygoods_n which = luxurygoods_n }
			subtract_variable = { which = staplefooditems_n which = staplefooditems_n }
			subtract_variable = { which = buildingsupplies_n which = buildingsupplies_n }
			subtract_variable = { which = militarysupplies_n which = militarysupplies_n }
		}
	}
}
