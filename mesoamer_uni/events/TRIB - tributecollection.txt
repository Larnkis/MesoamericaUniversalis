country_event = {
	id = tributecoll.02

	title = "tributecoll2.t"
	desc = "tributecoll2.d"
	picture = ECONOMY_eventPicture

	is_triggered_only = yes
	
	immediate = {
		every_owned_province = {
			subtract_variable = { which = prov_unrest which = prov_unrest }
			subtract_variable = { which = refusal_points which = refusal_points }
			remove_province_modifier = will_pay
			remove_province_modifier = wont_pay
		}
		every_owned_province = {
		
			#Adds Refusal Points From Unrest
			#1 Point for every 2 unrest
			if = {
				limit = {
					unrest = 0.01
				}
				export_to_variable = {
					which = prov_unrest 
					value = unrest
				}
				
				while = {
					limit = {
						check_variable = {
							which = prov_unrest 
							value = 0
						}
					}
					subtract_variable = {
						which = prov_unrest
						value = 0.01
					}
					change_variable = { which = refusal_points value = 0.005 }
				}
			}
			
			if = {
				limit = {
					owner = {
						has_idea = giftsoflove
					}
				}	
				change_variable = { 
					which = refusal_points
					value = -1
				}
			}	
			#Adds Refusal Points from Legitimacy 
			if = {
				limit = {
					owner = {
						NOT = { legitimacy = 20 }
					}
				}
				change_variable = { 
					which = refusal_points
					value = 1
				}
			}
			
			if = {
				limit = {
					owner = {
						NOT = { legitimacy = 40 }
					}
				}
				change_variable = { 
					which = refusal_points
					value = 1
				}
			}
			
			if = {
				limit = {
					owner = {
						NOT = { legitimacy = 60 }
					}
				}
				change_variable = { 
					which = refusal_points
					value = 1
				}
			}
			
			if = {
				limit = {
					owner = {
						NOT = { legitimacy = 80 }
					}
				}
				change_variable = { 
					which = refusal_points
					value = 1
				}
			}
			
			if = {
				limit = {
					owner = {
						NOT = { legitimacy = 100 }
					}
				}
				change_variable = { 
					which = refusal_points
					value = 1
				}
			}
			
			#Adds Refusal Points from Neighboring Provinces
			if = {
				limit = {
					any_neighbor_province = {
						check_variable = { 
							which = refusal_points 
							value = 10
						}
					}
				}
				change_variable = { which = refusal_points value = 4 }
			}
			
			#Pay or Naw
			add_permanent_province_modifier = {
				name = will_pay
				duration = -1
			}
			if = {
				limit = {
					check_variable = { #If refusal points are >= the cutoff, don't pay
						which = refusal_points
						which = REFUSAL_CUTOFF
					}
				}
				random_list = {
					50 = {
						add_country_modifier = {
							name = wont_pay
							duration = -1
						}
						remove_country_modifier = will_pay
					}
					50 = {
						#Do Nothing
					}
				}
			}
		}
	}

country_event = {
	id = tributecoll.03

	title = "tributecoll3.t"
	desc = "tributecoll3.d"
	picture = ECONOMY_eventPicture

	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			subtract_variable = { which = area_dev value = area_dev }
			
			every_owned_province = {
				limit = { #Do this to every province in the area that isn't the capital and has the will pay modifier
					area = western_shore_a ###REPLACE - AREA NAME###
					has_province_modifier = will_pay
					NOT = { is_capital = yes }
				}
				
				subtract_variable = { which = prov_dev_trib which = prov_dev_trib }
				
				export_to_variable = {
					which = prov_dev_trib
					value = development
				}
				
				while = { #Add this provinces development to the area's total
					limit = {
						check_variable = { 
							which = prov_dev_trib
							value = 0
						}
					}
					owner = {
						change_variable = { which = area_dev value = 1 }
					}
					subtract_variable = { which = prov_dev_trib value = 1 }
				}
				
			}
			
			#Convert the area's total development into a ducat value depending on tribute levels
			if = { 
				limit = {
					has_country_modifier = tribreqs_high
				}
				
				set_variable = { which = area_dev_mp which = area_dev }
				
				while = {
					limit = {
						check_variable = {
							which = area_dev_mp 
							value = 0 
						}
					}
					change_variable = { which = westshore_mp value = 1 } ####REPLACE - MP####
					subtract_variable = { 
						which = area_dev_mp
						value = 25
					}
				}
				
				while = {
					limit = {
						check_variable = {
							which = area_dev
							value = 0
						}
					}
					change_variable = { which = westshore_ad value = 1 } ####REPLACE - AD####
					subtract_variable = {
						which = area_dev
						value = 1
					}
				}
				
				multiply_variable = { 
					which = westshore_ad ####REPLACE - AD####
					value = 0.33
				}
				
			}
			
			if = { 
				limit = {
					has_country_modifier = tribreqs_normal 
				}
				
				set_variable = { which = area_dev_mp which = area_dev }
				
				while = {
					limit = {
						check_variable = {
							which = area_dev_mp 
							value = 0 
						}
					}
					change_variable = { which = westshore_mp value = 1 } ####REPLACE - MP####
					subtract_variable = { 
						which = area_dev_mp
						value = 35
					}
				}
				
				while = {
					limit = {
						check_variable = {
							which = area_dev
							value = 0
						}
					}
					change_variable = { which = westshore_ad value = 1 } ####REPLACE - AD####
					subtract_variable = {
						which = area_dev
						value = 1
					}
				}
				
				multiply_variable = { 
					which = westshore_ad ####REPLACE####
					value = 0.25
				}
				
			}	
			
			if = { 
				limit = {
					has_country_modifier = tribreqs_low
				}
				
				set_variable = { which = area_dev_mp which = area_dev }
				
				while = {
					limit = {
						check_variable = {
							which = area_dev_mp 
							value = 0 
						}
					}
					change_variable = { which = westshore_mp value = 1 } ####REPLACE - MP####
					subtract_variable = { 
						which = area_dev_mp
						value = 50
					}
				}
				
				while = {
					limit = {
						check_variable = {
							which = area_dev
							value = 0
						}
					}
					change_variable = { which = westshore_ad value = 1 } ####REPLACE - AD####
					subtract_variable = {
						which = area_dev
						value = 1
					}
				}
				
				multiply_variable = { 
					which = westshore_ad ####REPLACE - AD####
					value = 0.15
				}
				
			}
		}
	}
	
	option = { #Add ducats based on the value of [area name]_AreaDucats
		name = "tributecoll3.a"
		
		#Display Tooltips
		if = {
			limit = {
				has_country_modifier = tribreqs_high
			}
			custom_tooltip = TRIB_duc_high
			custom_tooltip = TRIB_mp_high
		}
		
		if = {
			limit = {
				has_country_modifier = tribreqs_normal
			}
			custom_tooltip = TRIB_duc_normal
			custom_tooltip = TRIB_mp_normal
		}
		
		if = {
			limit = {
				has_country_modifier = tribreqs_low
			}
			custom_tooltip = TRIB_duc_low
			custom_tooltip = TRIB_mp_low
		}
		
		while = { #Add Money
			limit = {
				check_variable = { 
					which = westshore_ad ####REPLACE - AD####
					value = 0
				}
			}
			subtract_variable = { 
				which = westshore_ad ####REPLACE - AD####
				value = 1
			}
			add_treasury = 1.0
		}	
		
		if = { #Add Manpower
			limit = {
				#is_at_war = yes
				NOT = { manpower_percentage = 1.5 }
			}
			while = {
				limit = {
					check_variable = {
						which = westshore_mp ####REPLACE - MP####
						value = 0
					}
				}
				subtract_variable = { 
					which = westshore_mp ####REPLACE - MP####
					value = 1
				}
				add_manpower = 1.0
			}	
		}
	}
}
