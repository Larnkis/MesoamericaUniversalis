#For Priestly Orders
country_event = {
	id = rpe.01
	
	title = "rpe1.t"
	desc = "rpe1.d"
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	#Political embargo
	option = {
		name = "rpe1.a"
		#custom_tooltip = "rpe1.ctt.1"
		trigger = {
			has_country_modifier = priestsingov
			NOT = { has_country_modifier = PO_LPE_cooldown }
		}
		remove_country_modifier = "priestsingov"
		add_country_modifier = {
			name = RPO_LPE_cooldown
			duration = 730
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_estate_influence_modifier = {
			estate = estate_church
			desc = POR_LPE
			influence = -25
			duration = -1
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = POR_EPLS
			influence = 10
			duration = -1
		}
	}

		#custom_tooltip = "rpe1.ctt.2"

			has_country_modifier = elevatedpriesthood

		}
		remove_country_modifier = elevatedpriesthood
		add_country_modifier = {
			name = RPO_ELS_cooldown
			duration = 730
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -5
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_estate_influence_modifier = {
			estate = estate_church
			desc = POR_EPLS
			influence = -20
			duration = -1
		}
		add_estate_influence_modifier = {
			estate = estate_nobles

			influence = 15
			duration = -1
		}
	}

			name = RPO_ATC_cooldown
