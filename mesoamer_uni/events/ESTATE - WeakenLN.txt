#Conflict Begins
country_event = {
	id = weakenln.02
	
	title = "weakenln2.t"
	desc = "weakenln2.d"
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "weakenln2.a"
		remove_country_modifier = ln_uprising_brewing
		add_stability = -1
		
		add_estate_loyalty = {
			estate = estate_lnobles
			loyalty = -100
		}
		
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = lower_noble_rebels
				win = yes
				size = 2
			}
		}
		
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = lower_noble_rebels
				win = yes
				size = 2
			}
		}
		
		every_owned_province = {
			limit = {
				has_estate = estate_lnobles
			}
			spawn_rebels = {
				type = lower_noble_rebels
				win = yes
				size = 2
			}
		}
		
		hidden_effect = {
			subtract_variable = { which = ln_favors which = ln_favors }
			clr_country_flag = ln_guarantee_noble_status
			clr_country_flag = ln_ban_new_men
			clr_country_flag = ln_increased_legal_protections
			clr_country_flag = ln_expanded_social_rights
			clr_country_flag = ln_exclusive_military_orders
		}
	}
}

#Win
country_event = {
	id = weakenln.03
	
	title = "weakenln3.t"
	desc = "weakenln3.d"
	picture = CONQUEST_NATIVE_eventPicture
	
	trigger = {
		has_disaster = ln_uprising
		NOT = { has_spawned_rebels = lower_noble_rebels }
	}
	
	mean_time_to_happen = {
		days = 5
	}
	
	option = {
		name = "weakenln3.a"
		
		end_disaster = ln_uprising
		
		add_years_of_income = -2.0
		
		add_estate_loyalty = {
			estate = estate_lnobles
			loyalty = 70
		}
		
		add_country_modifier = {
			name = weakened_ln
			duration = 9125
		}
		
		custom_tooltip = GOVS_mt_i_0250
		custom_tooltip = EST_fav_i_100
		hidden_effect = {
			change_variable = { which = an_favors value = 100 }
			change_variable = { which = meritocratictradition value = 25 }
		}
	}
	
	option = {
		name = "weakenln3.b"
		
		end_disaster = ln_uprising
		
		add_estate_loyalty = {
			estate = estate_lnobles
			loyalty = 50
		}

			duration = 9125
		}
		
		custom_tooltip = EST_fav_i_40
		hidden_effect = {
			change_variable = { which = an_favors value = 40 }
		}
	}
}
