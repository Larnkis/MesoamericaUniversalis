#For Nobles
country_event = {
	id = bdi.02

	title = "bdi2.t"
	desc = "bdi2.d"
	picture = CITY_VIEW_eventPicture

	is_triggered_only = yes 

	option = {
		name = "bdi2.a"
		trigger = {
			check_variable = { which = luxurygoods value = 20 }
			faction_influence = {
   				faction = nobles
    				influence = 10
			}
			NOT = {
				faction_influence = {
    					faction = nobles
   				        influence = 50
				}
			}
		}
		change_variable = { which = luxurygoods value = -20 }
		add_faction_influence = {
			faction = nobles
			influence = -2
		}
		add_faction_influence = {
			faction = currentdynasty
			influence = 1
		}
		country_event = { id = bdi.01 }
	}

	option = {
		name = "bdi2.b"
		trigger = {
			check_variable = { which = luxurygoods value = 50 }
			faction_influence = {
   				faction = nobles
    				influence = 10
			}
			NOT = {
				faction_influence = {
    					faction = nobles
   				        influence = 50
				}
			}
		}
		change_variable = { which = luxurygoods value = -50 }
		add_faction_influence = {
			faction = nobles
			influence = -5
		}
		add_faction_influence = {
			faction = currentdynasty
			influence = 2
		}
		country_event = { id = bdi.01 }
	}

	option = {
		name = "bdi2.c"
		trigger = {
			check_variable = { which = luxurygoods value = 100 }
			faction_influence = {
   				faction = nobles
    				influence = 10
			}
			NOT = {
				faction_influence = {
    					faction = nobles
   				        influence = 50
				}
			}
		}
		change_variable = { which = luxurygoods value = -100 }
		add_faction_influence = {
			faction = nobles
			influence = -8
		}
		add_faction_influence = {
			faction = currentdynasty
			influence = 4
		}
		country_event = { id = bdi.01 }
	}

	option = {
		name = "bdi2.dd"
		country_event = { id = bdi.01 }
	}
}
