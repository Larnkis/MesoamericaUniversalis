#For country it starts in
country_event = {
	id = institutions.01
	
	title = "institutions1.t"
	desc = "institutions1.d"
	
	picture = REFORM_eventPicture 
	
	is_triggered_only = yes
	
	option = {
		name = "institutions1.a"
		
		add_adm_power = 50
		add_dip_power = 50
		add_mil_power = 50
		add_prestige = 50
		custom_tooltip = WORLD_SYSTEM_PENALTY
		FROM = {
			add_permanent_province_modifier = {
				name = "birthplace_of_the_world_system"
				duration = -1
			}
		}
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { tag = ROOT }
				}
				country_event = { id = institutions.02 }
			}
			every_province = {
				limit = {
					region = central_mexico_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = southern_mexico_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = gulf_coast_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = michoacan_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = western_mexico_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = north_central_mexico_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = mayan_highlands_r
				}
				discover_country = ROOT
			}
			every_province = {
				limit = {
					region = yucatan_r
				}
				discover_country = ROOT
			}
		}
		
		ai_chance = {
			factor = 100
		}
	}
}
