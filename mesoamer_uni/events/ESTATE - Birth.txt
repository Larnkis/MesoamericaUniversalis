namespace = estatebirth

#Upper Nobles
country_event = {
	id = estatebirth.01
	
	title = "estatebirth1.t"
	desc = "estatebirth1.d"
	picture = COURT_eventPicture
	
	trigger = {
		ai = no
		government_rank = 2
		NOT = { has_country_flag = has_upper_nobles }
	}
	
	mean_time_to_happen = {
		months = 500
		
		modifier = {
			factor = 0.25
			government_rank = 3
		}
		
		modifier = {
			factor = 0.80
			NOT = { check_variable = { which = meritocratictradition value = 50 } }
		}
		
		modifier = {
			factor = 0.90
			treasury = 1000
		}
		
		modifier = {
			factor = 0.80
			treasury = 2000
		}
		
		modifier = {
			factor = 0.70
			treasury = 5000
		}
		
		modifier = {
			factor = 0.75
			num_of_cities = 10
		}
		
		modifier = {
			factor = 0.75
			num_of_protectorates = 10
		}
	}
	
	option = {
		name = "estatebirth1.a"
		custom_tooltip = EST_un_birth
		set_country_flag = has_upper_nobles
		clr_country_flag = has_altepetl_nobles
		
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_estate = estate_anobles
				}
				set_estate = estate_unobles
			}
		}
	}
}
