#Buy loyalty with luxury goods interfact
namespace = blg

#For Priests
country_event = {
	id = blg.01
	
	title = "blg1.t"
	desc = "blg1.d"
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	
	#Small gift
	option = {
		name = "blg1.a"
		trigger = { 
			check_variable = { which = luxurygoods which = blcpriestslow }
		}
		subtract_variable = { which = luxurygoods which = blcpriestslow }
		
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 5
		}
	}
	
	#Medium gift
	option = {
		name = "blg1.b"
		trigger = { 
			check_variable = { which = luxurygoods which = blcpriests }
		}
		subtract_variable = { which = luxurygoods which = blcpriests }
		
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}
	}
	
	#Large gift
	option = {
		name = "blg1.c"
		trigger = { 
			check_variable = { which = luxurygoods which = blcpriestshigh }
		}
		subtract_variable = { which = luxurygoods which = blcpriestshigh }
		
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 15
		}
	}
	
	option = {
		name = "blg1.dd"
	}
}
