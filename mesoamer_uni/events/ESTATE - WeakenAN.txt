#Disaster Begins
country_event = {
	id = weakenan.02
	
	title = "weakenan2.t"
	desc = "weakenan2.d"
	picture = CIVIL_WAR_eventPicture
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "weakenan2.a"
		remove_country_modifier = an_uprising_brewing
		add_stability = -1
		
		add_estate_loyalty = {
			estate = estate_anobles
			loyalty = -100
		}
		
		capital_scope = {
			spawn_rebels = {
				type = altepetl_noble_rebels
				size = 4
			}
		}
		
		every_owned_province = {
			limit = {
				has_estate = estate_anobles
			}
			spawn_rebels = {
				type = altepetl_noble_rebels
				win = yes
				size = 2
			}
		}
		
		hidden_effect = {
			subtract_variable = { which = an_favors which = an_favors }
			clr_country_flag = an_exclusive_military_leadership
			clr_country_flag = an_exclusive_military_orders
			clr_country_flag = an_exclusive_legal_protections
			clr_country_flag = an_exclusive_bureaucracy_positions
			clr_country_flag = an_exclusive_social_rights
			clr_country_flag = an_share_tribute
		}
	}
}

#Win
country_event = {
	id = weakenan.03
	
	title = "weakenan3.t"
	desc = "weakenan3.d"
	picture = CONQUEST_NATIVE_eventPicture
	
	trigger = {
		has_disaster = an_uprising
		NOT = { has_spawned_rebels = altepetl_noble_rebels }
	}
	
	mean_time_to_happen = {
		days = 5
	}
	
	option = { 
		name = "weakenan3.a"
		
		end_disaster = an_uprising
		
		add_years_of_income = -2.0
		
		add_estate_loyalty = {
			estate = estate_anobles
			loyalty = 70
		}
		
		add_country_modifier = {
			name = weakened_an
			duration = 9125
		}
		
		custom_tooltip = GOVS_mt_i_0250
		custom_tooltip = EST_fav_i_100
		hidden_effect = {
			change_variable = { which = an_favors value = 100 }
			change_variable = { which = meritocratictradition value = 25 }
		}
	}
	
	option = {
		name = "weakenan3.b"
		
		end_disaster = an_uprising
		
		add_estate_loyalty = {
			estate = estate_anobles
			loyalty = 50
		}
		
		add_country_modifier = {
			name = weakened_an
			duration = 9125
		}
		
		custom_tooltip = EST_fav_i_40
		hidden_effect = {
			change_variable = { which = an_favors value = 40 }
		}
	}
}
